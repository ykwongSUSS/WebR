2.2 Alternatively, create data.table using combined vectors, lists

ID2 <- str_c("SD",101:110)
Gender2 <- c("M","F","F","F","M","F","M","M","M","M")
Test21 <- c(11,8,13,15,10,11,6,12,10,7)
Test22 <- c(48,43,15,20,40,23,24,47,21,27)
Exam23 <- c(61,77,49,57,82,65,46,80,48,62)
dt2 <- data.table(ID2,Gender2,Test21,Test22,Exam23)
class(dt2)
[1] "data.table" "data.frame"

2.6 Using cut to data segment and assign letter grade

breaks<- c(0,40,44,49,54,59,64,69,74,79,84,100)
labels<-c("F","D","D+","C","C+","B-","B","B+","A-","A","A+")
lgrades<-cut(score, breaks, labels, right=TRUE)
dt[,Grade := lgrades]
