1. Install data.table

library(data.table)

2. Create a data.table from lists

list_1 <- c("a","b","c","d")
list_2 <- c("1","2","3","4")
list_3 <- c("aa","bb","cc","dd")
dt2 <- data.table(v1=list_1,v2=list_2,v3=list_3)
